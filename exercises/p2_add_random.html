from js import document, localStorage, window
import random, re, sys, io

# Randomize the name and the new gem for this run
names = ["Kai", "Ava", "Milo", "Zoe", "Lia", "Noah", "Ivy", "Owen", "Maya", "Leo"]
gems  = ["ruby", "diamond", "sapphire", "emerald", "garnet", "opal", "pearl"]

NAME = random.choice(names)
NEW  = random.choice(gems)

document.getElementById('nm').textContent  = NAME
document.getElementById('gem').textContent = NEW

def gi(k):
    v = localStorage.getItem(k)
    try:
        return int(v) if v is not None else 0
    except Exception:
        return 0

def setb(stars=None, bugs=None):
    if stars is not None: localStorage.setItem('stars', str(stars))
    if bugs is not None: localStorage.setItem('bugs', str(bugs))
    document.getElementById('stars').textContent = f'‚≠ê Stars: {gi("stars")}'
    document.getElementById('bugs').textContent  = f'üêû Bugs: {gi("bugs")}'

def run(evt):
    ns = {}
    src = document.getElementById('editor').value
    out_el = document.getElementById('out')

    f = io.StringIO()
    try:
        # Compile user's code and call their function
        exec(src, ns)
        assert 'chest_report' in ns, "Make a function named chest_report(name, chest, new_item)."

        chest = ["ruby", "diamond"]  # initial chest
        old_stdout = sys.stdout
        sys.stdout = f
        try:
            ns['chest_report'](NAME, chest, NEW)
        finally:
            sys.stdout = old_stdout

        output = f.getvalue()

        # 1) Hello, NAME! (uppercase)
        simplified = re.sub(r'[^A-Z ]', '', output.upper()).replace(',', '')
        assert ("HELLO " + NAME.upper()) in simplified, "Say 'Hello, NAME!' (NAME in UPPERCASE)."

        # 2) Append happened
        assert chest[-1] == NEW, "Did you append the new gem to the chest?"

        # 3) Printed first and last items
        assert (chest[0] in output) and (chest[-1] in output), "Print first and last items of the chest."

        out_el.textContent = "All checks passed! ‚≠ê"
        setb(stars = gi('stars') + 1)
        try:
            window.confetti()
        except Exception:
            pass

    except Exception as e:
        out_el.textContent = "Bug found: " + str(e)
        setb(bugs = gi('bugs') + 1)

document.getElementById('run').addEventListener('click', run)

# Buttons to fill starter and solution
STARTER = '''def chest_report(name, chest, new_item):
    # 1) say hello in UPPERCASE (use name.upper())
    # 2) add the new_item to the chest (chest.append(...))
    # 3) print the first and last item (print(chest[0], chest[-1]))
    pass
'''

SOLUTION = '''def chest_report(name, chest, new_item):
    print("Hello, " + name.upper() + "!")
    chest.append(new_item)
    print(chest[0], chest[-1])
'''

def fill_start(evt):
    document.getElementById('editor').value = STARTER

def fill_solution(evt):
    document.getElementById('editor').value = SOLUTION

document.getElementById('fill-start').addEventListener('click', fill_start)
document.getElementById('fill-solution').addEventListener('click', fill_solution)
